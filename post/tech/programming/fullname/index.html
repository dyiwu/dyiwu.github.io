$ cat fullname.awk
#!/usr/bin/awk -f
{
  OFS="";
  $1=$1;
  $2=$2;
  $3=$3;
#  print "---",$1,"---,"$2,"---,"$3"---";
  if ($1 == "" && $2 == "" && $3 == "") print ;
  else
  {
  FirstName=$1;
  if ($3 != "")
    { MiddleName=$2;
      LastName=$3;
      print LastName,", ",FirstName," ",MiddleName;
      }
  else
    { LastName=$2;
      print LastName,", ",FirstName;
      }
  }
}
